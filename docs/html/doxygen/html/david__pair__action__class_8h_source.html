66  virtual double CalcU(double r, double r_p, double s, const uint32_t level)

90  double s_2_k = s_2;

95  // indexing into the 2darray

97  u += (u_cof)*z_2_j*current_s;
98  z_2_j *= z_2;
99  current_s *= i_s_2;
100  }
101  s_2_k *= s_2;
102  }

133  virtual double CalcdUdBeta(double r, double r_p, double s, const uint32_t level)
134  {
135  // Constants
136  double q = 0.5*(r + r_p);
137  double z = r - r_p;
138 
139  // Limits
140  double r_min, r_max;
141  GetLimits(r_min, r_max, r, r_p, grid);
142 
143  // This is the endpoint action
144  double u, v, du;
145  vec<double> r_vals(n_val+1), r_p_vals(n_val+1);
146  eval_multi_NUBspline_1d_d(u_kj(level),r,r_vals.memptr());
147  eval_multi_NUBspline_1d_d(u_kj(level),r_p,r_p_vals.memptr());
148  v = 0.5*(r_vals(0) + r_p_vals(0));
149  u = 0.5*(r_vals(1) + r_p_vals(1));
150  eval_multi_NUBspline_1d_d(du_kj_dbeta(level),r,r_vals.memptr());
151  eval_multi_NUBspline_1d_d(du_kj_dbeta(level),r_p,r_p_vals.memptr());
152  du = 0.5*(r_vals(1) + r_p_vals(1));
153 
154  // Compensate for potential, which is subtracted from diaganal action in dm file.
155  du += v;
156 
157  // Add in off-diagonal terms
158  if (s > 0.0 && q<r_max) {
159  vec<double> u_q_vals(n_val+1), du_q_vals(n_val+1);
160  eval_multi_NUBspline_1d_d(u_kj(level),q,u_q_vals.memptr());
161  eval_multi_NUBspline_1d_d(du_kj_dbeta(level),q,du_q_vals.memptr());
162  double z_2 = z*z;
163  double s_2 = s*s;
164  double i_s_2 = 1./s_2;
165  double s_2_k = s_2;
166  for (uint32_t k=1; k<=n_order; k++) {
167  double z_2_j = 1;
168  double current_s = s_2_k;
169  for (uint32_t j=0; j<=k; j++){
170  // indexing into the 2darray
171  double u_cof = u_q_vals(k*(k+1)/2+j+1);
172  double du_cof = du_q_vals(k*(k+1)/2+j+1);
173  u += (u_cof)*z_2_j*current_s;
174  du += (du_cof)*z_2_j*current_s;
175  z_2_j *= z_2;
176  current_s *= i_s_2;
177  }
178  s_2_k *= s_2;
179  }
180  }
