39  virtual bool Attempt()
40  {
41  int p_i = rng.UnifRand(n_part) - 1; // Pick particle at random
42  bead0 = rng.UnifRand(path.n_bead) - 1; // Pick first bead at random
43  bead1 = bead0 + n_bisect_beads; // Set last bead in bisection
44  bool roll_over = bead1 > (path.n_bead-1); // See if bisection overflows to next particle
45  bool include_ref = path.species_list[species_i]->fixed_node &&
46  ((bead0<=path.ref_bead && bead1>=path.ref_bead) ||
47  (roll_over && path.bead_loop[bead1]>=path.ref_bead));
48  if (include_ref)
49  ref_attempt++;
50 
51  // Set up pointers
52  std::shared_ptr<Bead> bead_i(path(species_i,p_i,bead0));
53  std::shared_ptr<Bead> bead_f(bead_i);
54  affected_beads.clear();
55  affected_beads.push_back(bead_i);
56  for (uint32_t i=0; i<n_bisect_beads; ++i) {
57  bead_f = bead_f->next;
58  affected_beads.push_back(bead_f);
59  }
60 
61  // Set which particles are affected by the move
62  std::vector<std::pair<uint32_t,uint32_t>> particles;
63  particles.push_back(std::make_pair(species_i,p_i));
64  if (bead_f->p != p_i) // fixme: may be overkill
65  particles.push_back(std::make_pair(species_i,bead_f->p));
66 
67  // Perform the bisection (move exactly through kinetic action)
68  std::shared_ptr<Bead> bead_a, bead_b, bead_c;
69  double prev_action_change = 0.;
70  for (int level_i = n_level-1; level_i >= 0; level_i -= 1) {
71  uint32_t skip = 1<<level_i;
72  double level_tau = path.tau*skip;
73  double sigma2 = lambda*level_tau;
74  double sigma = sqrt(sigma2);
75 
76  double old_log_sample_prob = 0.;
77  double new_log_sample_prob = 0.;
78  bead_a = bead_i;
79  while(bead_a != bead_f) {
80  // Set beads
81  bead_b = path.GetNextBead(bead_a,skip);
82  bead_c = path.GetNextBead(bead_b,skip);
83 
84  // Old sampling
85  path.SetMode(OLD_MODE);
86  vec<double> delta_old(path.Dr(bead_b,path.RBar(bead_c,bead_a)));
87  old_log_sample_prob += rho_free_splines[level_i].GetLogRhoFree(delta_old);
88 
89  // New sampling
90  path.SetMode(NEW_MODE);
91  vec<double> delta_new(path.n_d);
92  rng.NormRand(delta_new, 0., sigma);
93  path.PutInBox(delta_new);
94  bead_b->r = path.RBar(bead_c, bead_a) + delta_new;
95  new_log_sample_prob += rho_free_splines[level_i].GetLogRhoFree(delta_new);
96 
97  // Advance beads
98  bead_a = bead_c;
99  }
100 
101  // Calculate action change
102  double old_action = 0.;
103  double new_action = 0.;
104  for (auto& action: action_list) {
105  // Old action
106  path.SetMode(OLD_MODE);
107  old_action += action->GetAction(bead0, bead1, particles, level_i);
108 
109  // New action
110  path.SetMode(NEW_MODE);
111  new_action += action->GetAction(bead0, bead1, particles, level_i);
112  }
113 
114  double log_sample_ratio = -new_log_sample_prob + old_log_sample_prob;
115  double current_action_change = new_action - old_action;
116  double log_accept_probability = log_sample_ratio - current_action_change + prev_action_change;
117 
118  // Metropolis reject step
119  if (log_accept_probability < log(rng.UnifRand()))
120  return 0;
121 
122  prev_action_change = current_action_change;
123  }
124 
125  if (include_ref)
126  ref_accept++;
127 
128  return 1;
129  }
