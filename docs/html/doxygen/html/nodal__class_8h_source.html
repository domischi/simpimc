83  double LineSearchDistance(const int b_i, const std::vector<std::shared_ptr<Bead>> &ref_b, const std::vector<std::shared_ptr<Bead>> &other_b_i)
84  {
85  // Calculate determinant and gradiant
86  SetRhoFGradRhoF(b_i,ref_b,other_b_i);
87  if (rho_f(b_i) < 0.)
88  return -1.;
89 
90  // Set up temporary storage fields
91  field<vec<double>> other_b_i_0(n_part);
92  for (uint32_t p_i=0; p_i<n_part; ++p_i)
93  other_b_i_0(p_i) = path.GetR(other_b_i[p_i]);
94 
95  // Set initial gradient unit vector
96  field<vec<double>> grad_rho_f_0(grad_rho_f.row(b_i));
97  double grad_mag(FieldVecMag(grad_rho_f_0));
98  double rho_f_0 = rho_f(b_i);
99  double dist_0 = rho_f_0/grad_mag;
100  double dist_0_i_grad_mag = dist_0/grad_mag;
101  double abs_dist_0 = fabs(dist_0);
102 
103  // Find close point on other side of the node (along this distance of the gradient)
104  bool done = false;
105  double min_factor(0.), max_factor(0.5), max_dist(MaxDistance(b_i,ref_b,other_b_i));
106  while ((rho_f_0*rho_f(b_i) > 0.) && (max_factor*dist_0 < max_dist)) {
107  max_factor *= 2.;
108  for (uint32_t p_i=0; p_i<n_part; ++p_i)
109  path.GetR(other_b_i[p_i]) = other_b_i_0(p_i) - max_factor*dist_0_i_grad_mag*grad_rho_f_0(p_i);
110  SetRhoF(b_i,ref_b,other_b_i);
111  }
112 
113  // Perform a bisection search for the sign change
114  double dist_b_i(max_dist);
115  if (rho_f_0*rho_f(b_i) < 0.) {
116  double try_factor(1.);
117  while (((max_factor-min_factor)*abs_dist_0>dist_tolerance) && (min_factor*abs_dist_0<max_dist)) {
118  try_factor = 0.5*(max_factor+min_factor);
119  for (uint32_t p_i=0; p_i<n_part; ++p_i)
120  path.GetR(other_b_i[p_i]) = other_b_i_0(p_i) - try_factor*dist_0_i_grad_mag*grad_rho_f_0(p_i);
121  SetRhoF(b_i,ref_b,other_b_i);
122  if (rho_f_0*rho_f(b_i) > 0.)
123  min_factor = try_factor;
124  else
125  max_factor = try_factor;
126  }
127  if (min_factor*abs_dist_0 > max_dist)
128  dist_b_i = max_dist;
129  else
130  dist_b_i = abs_dist_0*try_factor;
131  }
132 
133  // Reset positions and determinant
134  for (uint32_t p_i=0; p_i<n_part; ++p_i)
135  path.GetR(other_b_i[p_i]) = other_b_i_0(p_i);
136  rho_f(b_i) = rho_f_0;
137 
138  // Track number of complete line search calls
139  n_line_search++;
140 
141  return dist_b_i;
142  }

145  double MaxDistance(const int b_i, const std::vector<std::shared_ptr<Bead>> &ref_b, const std::vector<std::shared_ptr<Bead>> &other_b_i)
146  {
147  double dist_b_i = 1./dist_tolerance;

152  return dist_b_i;

156  double HyperplaneDistance(const int b_i, const std::vector<std::shared_ptr<Bead>> &ref_b, const std::vector<std::shared_ptr<Bead>> &other_b_i)
157  {
158  double dist_b_i = 1./dist_tolerance;
159  for (uint32_t p_i=0; p_i<n_part-1; ++p_i) {
160  for (uint32_t p_j=p_i+1; p_j<n_part; ++p_j) {
161  vec<double> r_tau_0 = path.GetR(path(species_i,p_i,b_i));
162  vec<double> r_tau_1 = path.GetR(path(species_i,p_j,b_i));
163  vec<double> r_star_0 = path.GetR(ref_b[p_i]);
164  vec<double> r_star_1 = path.GetR(ref_b[p_j]);
165  vec<double> r_tau_0_p(r_tau_0);
166  r_tau_0_p(1) = r_tau_1(0) - (r_tau_0(1)-r_tau_1(1))*(r_star_0(1)-r_star_1(1))/(r_star_0(0)-r_star_1(0));
167  double a, b(-1.), c;
168  GetLine(r_tau_0_p, r_tau_1, a, c);
169  dist_b_i = fabs(a*r_tau_0(0) + b*r_tau_0(1) + c)/sqrt(a*a + b*b);
170  dist_b_i -= nearbyint(dist_b_i*path.iL)*path.L;
171  dist_b_i = fabs(dist_b_i);
172  }
173  }
174  return dist_b_i*M_SQRT2;
175  }

178  double NewtonRaphsonDistance(const int b_i, const std::vector<std::shared_ptr<Bead>> &ref_b, const std::vector<std::shared_ptr<Bead>> &other_b_i)
179  {
180  // Calculate determinant and gradiant
181  SetRhoFGradRhoF(b_i,ref_b,other_b_i);
182  if (rho_f(b_i) < 0.)
183  return -1.;
184  if (std::isnan(grad_rho_f(b_i,0)(0)))
185  return LineSearchDistance(b_i,ref_b,other_b_i);
186 
187  // Set up temporary storage fields
188  field<vec<double>> t_other_b_i(n_part), other_b_i_0(n_part);
189  for (uint32_t p_i=0; p_i<n_part; ++p_i) {
190  other_b_i_0(p_i) = path.GetR(other_b_i[p_i]);
191  t_other_b_i(p_i) = path.GetR(other_b_i[p_i]);
192  }
193 
194  // Set initial gradient unit vector
195  field<vec<double>> grad_rho_f_0(grad_rho_f.row(b_i));
196  double grad_mag(FieldVecMag(grad_rho_f_0));
197  if (grad_mag < dist_tolerance)
198  return LineSearchDistance(b_i,ref_b,other_b_i);
199  double rho_f_0 = rho_f(b_i);
200  double dist_0 = rho_f_0/grad_mag;
201 
202  // Perform Newton-Raphson steps
203  bool done = false;
204  int i_dist_step = 0;
205  while (!done && (i_dist_step < max_dist_steps)) {
206  // Find zero crossing with current gradient
207  double t_dist = 2.*rho_f(b_i)/grad_mag;
208  do {
209  t_dist *= 0.5;
210  for (uint32_t p_i=0; p_i<n_part; ++p_i) {
211  path.GetR(other_b_i[p_i]) = t_other_b_i(p_i) - (t_dist/grad_mag)*grad_rho_f(b_i,p_i);
212  }
213  SetRhoF(b_i,ref_b,other_b_i);
214  } while (rho_f(b_i) < 0.);
215 
216  // Set new temporary path
217  for (uint32_t p_i=0; p_i<n_part; ++p_i)
218  t_other_b_i(p_i) = path.GetR(other_b_i[p_i]);
219 
220  // Check if tolerance is satisfied
221  if (t_dist < dist_tolerance*dist_0)
222  done = true;
223 
224  // Set new determinant and gradient
225  SetRhoFGradRhoF(b_i,ref_b,other_b_i);
226  if (!std::isnan(grad_rho_f(b_i,0)(0))) {
227  grad_mag = FieldVecMag(grad_rho_f,b_i);
228  if (grad_mag > 0.) {
229  // Iterate and continue
230  i_dist_step++;
231  continue;
232  }
233  }
234 
235  // Reset positions, determinant, and gradient
236  for (uint32_t p_i=0; p_i<n_part; ++p_i)
237  path.GetR(other_b_i[p_i]) = other_b_i_0(p_i);
238  grad_rho_f.row(b_i) = grad_rho_f_0;
239  rho_f(b_i) = rho_f_0;
240  return LineSearchDistance(b_i,ref_b,other_b_i);
241 
242  }
243 
244  // Compute distance
245  for (uint32_t p_i=0; p_i<n_part; ++p_i) {
246  t_other_b_i(p_i) -= other_b_i_0(p_i);
247  path.PutInBox(t_other_b_i(p_i));
248  }
249  double dist_b_i(FieldVecMag(t_other_b_i));
250 
251  // Reset positions, determinant, and gradient
252  for (uint32_t p_i=0; p_i<n_part; ++p_i)
253  path.GetR(other_b_i[p_i]) = other_b_i_0(p_i);
254  grad_rho_f.row(b_i) = grad_rho_f_0;
255  rho_f(b_i) = rho_f_0;
256 
257  // Track number of complete Newton Raphson calls
258  n_newton_raphson++;
259 
260  return dist_b_i;
261  }

285  void SetRhoFGradRhoF(const int b_i, const std::vector<std::shared_ptr<Bead>> &ref_b, const std::vector<std::shared_ptr<Bead>> &other_b_i)
286  {
287  // Set slice distance
288  uint32_t abs_slice_diff = abs(path.bead_loop(b_i)-path.ref_bead);
289  uint32_t min_slice_diff = std::min(abs_slice_diff, path.n_bead-abs_slice_diff);
290 
291  // Compute Gij matrix and its gradient
292  mat<double> g_ij(n_part,n_part);
293  field<vec<double>> dg_ij_dr(n_part,n_part);
294  for (uint32_t p_i=0; p_i<n_part; ++p_i) {
295  for (uint32_t p_j=0; p_j<n_part; ++p_j) {
296  dg_ij_dr(p_i,p_j).zeros(path.n_d);
297  g_ij(p_i,p_j) = GetGijDGijDr(ref_b[p_i], other_b_i[p_j], min_slice_diff, dg_ij_dr(p_i,p_j));
298  }
299  }
300 
301  // Compute determinant with cofactors using Gauss-Jordan elimination
302  rho_f(b_i) = det(g_ij);
303  if (rho_f(b_i) < 0.) // Check sign
304  return;
305 
306  // Create cofactor matrix
307  mat<double> i_g_ij;
308  try {
309  i_g_ij = inv(g_ij);
310  } catch (const std::runtime_error& error) {
311  rho_f(b_i) = -1.;
312  grad_rho_f(b_i,0) = sqrt(-1.);
313  return;
314  }
315 
316  mat<double> cofactors_g_ij(rho_f(b_i)*trans(i_g_ij)); // TODO: Should just compute directly from adjugate
317 
318  // Compute gradient of determinant
319  for (uint32_t p_i=0; p_i<n_part; ++p_i) {
320  grad_rho_f(b_i,p_i).zeros();
321  for (uint32_t p_j=0; p_j<n_part; ++p_j)
322  grad_rho_f(b_i,p_i) -= dg_ij_dr(p_j,p_i)*cofactors_g_ij(p_j,p_i);
323  }

352  double DistanceAction(const std::vector<uint32_t> &b_i_vec, const std::vector<std::shared_ptr<Bead>> &ref_b, const std::vector<std::vector<std::shared_ptr<Bead>>> &other_b, const int n_bead_in_move, const bool check_all)
353  {
354  // Set nodal distances
355  double tot = 0.;
356  vec<double> t_dist(n_bead_in_move);
357  for (uint32_t b_i=0; b_i<n_bead_in_move; ++b_i) {
358  if (b_i_vec[b_i] != path.ref_bead) {
359  t_dist(b_i) = GetNodalDistance(b_i_vec[b_i],ref_b,other_b[b_i]);
360  if (t_dist(b_i) < 0.) {
361  tot += 1.e100;
362  break;
363  }
364  }
365  }
366 
367  // Compute action from nodal distance
368  if (tot == 0.) {
369  uint32_t skip = b_i_vec[1] - b_i_vec[0];
370  double i_lambda_level_tau = 1./(path.species_list[species_i]->lambda*skip*path.tau);
371  for (uint32_t b_i=0; b_i<n_bead_in_move-1; ++b_i) {
372 
373  // Set slice variables
374  uint32_t b_j = b_i + 1;
375  bool b_i_is_ref = b_i_vec[b_i] == path.ref_bead;
376  bool b_j_is_ref = b_i_vec[b_j] == path.ref_bead;
377 
378  // Compute action
379  if (!b_i_is_ref && (t_dist(b_i)<0.)) {
380  tot = 1.e100;
381  break;
382  } else if (!b_j_is_ref && (t_dist(b_j)<0.)) {
383  tot = 1.e100;
384  break;
385  } else if (b_i_is_ref || (t_dist(b_i)==0.))
386  tot -= log1p(-exp(-t_dist(b_j)*t_dist(b_j)*i_lambda_level_tau));
387  else if (b_j_is_ref || (t_dist(b_j)==0.))
388  tot -= log1p(-exp(-t_dist(b_i)*t_dist(b_i)*i_lambda_level_tau));
389  else
390  tot -= log1p(-exp(-t_dist(b_i)*t_dist(b_j)*i_lambda_level_tau));
391 
392  // Store if first time or in new mode
393  if (path.GetMode()==NEW_MODE || is_first_time[b_i_vec[b_i]]) {
394  dist(b_i_vec[b_i]) = t_dist(b_i);
395  is_first_time[b_i_vec[b_i]] = false;
396  }
397  if (path.GetMode()==NEW_MODE || is_first_time[b_i_vec[b_j]]) {
398  dist(b_i_vec[b_j]) = t_dist(b_j);
399  is_first_time[b_i_vec[b_j]] = false;
400  }
401 
402  }
403  }

512  virtual double DActionDBeta()
513  {
514  // Zero if not using the nodal distance or using it as an importance weight
515  if (!use_nodal_distance)
516  return 0.;
517 
518  // Initialize other beads
519  std::vector<std::shared_ptr<Bead>> ref_b(n_part);
520  std::vector<std::vector<std::shared_ptr<Bead>>> other_b(path.n_bead);
521  for (uint32_t p_i=0; p_i<n_part; ++p_i)
522  ref_b[p_i] = path(species_i,p_i,path.ref_bead);
523  for (uint32_t p_i=0; p_i<n_part; ++p_i)
524  other_b[0].push_back(path.GetPrevBead(ref_b[p_i],path.ref_bead));
525 
526  std::vector<uint32_t> b_i_vec;
527  b_i_vec.push_back(other_b[0][0]->b);
528  for (uint32_t b_i=1; b_i<path.n_bead; ++b_i) {
529  for (uint32_t p_i=0; p_i<n_part; ++p_i)
530  other_b[b_i].push_back(path.GetNextBead(other_b[b_i-1][p_i],1));
531  b_i_vec.push_back(other_b[b_i][0]->b);
532  }
533 
534  // Compute action from nodal distance
535  double tot = 0.;
536  double i_lambda_tau = 1./(path.species_list[species_i]->lambda*path.tau);
537  for (uint32_t b_i=0; b_i<path.n_bead; ++b_i) {
538  // Set slice variables
539  uint32_t b_j = path.bead_loop(b_i + 1);
540  bool b_i_is_ref(b_i == path.ref_bead);
541  bool b_j_is_ref(b_j == path.ref_bead);
542 
543  // Compute dist2
544  double dist_b_i, dist_b_j, dist2;
545  if (!b_i_is_ref)
546  dist_b_i = dist(b_i); //GetNodalDistance(b_i_vec[b_i],ref_b,other_b[b_i]);
547  if (!b_j_is_ref)
548  dist_b_j = dist(b_j); //GetNodalDistance(b_i_vec[b_j],ref_b,other_b[b_j]);
549  if (b_i_is_ref || (dist_b_i==0.))
550  dist2 = dist_b_j*dist_b_j;
551  else if (b_j_is_ref || (dist_b_j==0.))
552  dist2 = dist_b_i*dist_b_i;
553  else
554  dist2 = dist_b_i*dist_b_j;
555 
556  // Compute action
557  double dist2_i_lambda_tau = dist2*i_lambda_tau;
558  double exp_dist2_i_lambda_tau_m1 = expm1(dist2_i_lambda_tau);
559  if (std::fpclassify(exp_dist2_i_lambda_tau_m1) == FP_NORMAL)
560  tot += dist2_i_lambda_tau/(path.tau*exp_dist2_i_lambda_tau_m1);
561  }
562 
563  return tot/path.n_bead;
564  }

567  virtual double GetAction(const uint32_t b0, const uint32_t b1, const std::vector<std::pair<uint32_t,uint32_t>> &particles, const uint32_t level)
568  {
569  // No need to check old action if not using the nodal distance
570  if (path.GetMode()==OLD_MODE && !use_nodal_distance)
571  return 0.;
572 
573  // Decide whether or not to check the node
574  if (level > max_level || !path.species_list[species_i]->fermi)
575  return 0.;
576  bool check_node = false;
577  for (auto& p: particles) {
578  if (p.first == species_i) {
579  check_node = true;
580  break;
581  }
582  }
583  if (!check_node)
584  return 0.;
585 
586  // See if ref slice included
587  bool check_all = false;
588  if (b1 < path.n_bead)
589  check_all = ((b0 <= path.ref_bead) && (b1 >= path.ref_bead));
590  else
591  check_all = (path.bead_loop(b1) >= path.ref_bead);
592 
593  // Set start and end beads
594  if (check_all) {
595  start_b = 0;
596  end_b = path.n_bead-1;
597  } else {
598  start_b = b0;
599  end_b = b1;
600  }
601 
602  // Set up bead index vector
603  uint32_t skip = 1<<level;
604  int n_bead_in_move = 1 + (end_b - start_b)/skip;
605  std::vector<uint32_t> b_i_vec;
606  for (uint32_t b_i=start_b; b_i<=end_b; b_i+=skip)
607  b_i_vec.push_back(path.bead_loop(b_i));
608 
609  // Initialize other beads
610  std::vector<std::shared_ptr<Bead>> ref_b(n_part);
611  std::vector<std::vector<std::shared_ptr<Bead>>> other_b(n_bead_in_move);
612  int slice_diff_0 = path.bead_loop(start_b) - path.ref_bead;
613  int abs_slice_diff_0 = abs(slice_diff_0);
614  for (uint32_t p_i=0; p_i<n_part; ++p_i)
615  ref_b[p_i] = path(species_i,p_i,path.ref_bead);
616  if (slice_diff_0 >= 0) {
617  for (uint32_t p_i=0; p_i<n_part; ++p_i)
618  other_b[0].push_back(path.GetNextBead(ref_b[p_i],abs_slice_diff_0));
619  } else {
620  for (uint32_t p_i=0; p_i<n_part; ++p_i)
621  other_b[0].push_back(path.GetPrevBead(ref_b[p_i],abs_slice_diff_0));
622  }
623  for (uint32_t b_i=1; b_i<n_bead_in_move; ++b_i)
624  for (uint32_t p_i=0; p_i<n_part; ++p_i)
625  other_b[b_i].push_back(path.GetNextBead(other_b[b_i-1][p_i],skip));
626 
627  // Decide which type of action to compute
628  if (use_nodal_distance)
629  return DistanceAction(b_i_vec, ref_b, other_b, n_bead_in_move, check_all);
630  else
631  return SimpleAction(b_i_vec, ref_b, other_b, n_bead_in_move, check_all);
632  }
