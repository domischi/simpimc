227  void SetupKs(const double k_cut)
228  {
229  if (k_cut <= k_c)
230  return;
231  else {
232  ks.clear();
233  k_indices.clear();
234  mag_ks.clear();

238  // Calculate k box
239  k_box.set_size(n_d);
240  for (uint32_t d_i=0; d_i<n_d; d_i++)
241  k_box(d_i) = 2.*M_PI/L;
242 
243  // Calculate max k index based on k cutoff
244  max_k_index.set_size(n_d);
245  for (uint32_t d_i=0; d_i<n_d; d_i++)
246  max_k_index(d_i) = (uint32_t) ceil(1.1*k_cut/k_box(d_i));
247 
248  // Set up C vector
249  c_k.set_size(n_d);
250  for (uint32_t d_i=0; d_i<n_d; d_i++)
251  c_k(d_i).set_size(2*max_k_index(d_i)+1);
252 
253  // Generate all possible combinations and permutations of k indices
254  std::vector<int> indices;
255  for (uint32_t d_i=0; d_i<n_d; d_i++)
256  for (int i=-max_k_index(d_i); i<=max_k_index(d_i); i++)
257  indices.push_back(i);
258  std::vector<std::vector<int>> tmp_k_indices;
259  GenCombPermK(tmp_k_indices, indices, n_d, false, true);
260 
261  // Iterate through indices, form k vectors, and include only those that should be included
262  for (auto& ik: tmp_k_indices) {
263  vec<double> k(n_d);
264  vec<int> ki(n_d);
265  for (uint32_t d_i=0; d_i<n_d; d_i++) {
266  k(d_i) = ik[d_i]*k_box(d_i);
267  ki(d_i) = max_k_index(d_i) + ik[d_i];
268  }
269  if (Include(k, k_cut)) {
270  ks.push_back(k);
271  k_indices.push_back(ki);

273  }

421  std::complex<double> tmp_c_k;

545  void SetupPermSectors(const uint32_t s_i, const uint32_t sectors_max)
546  {
547  int n = species_list[s_i]->n_part;
548  if (!perm_sectors_setup[s_i]) {
549  std::cout << "Setting up permutation sectors..." << std::endl;
550  std::vector<int> a;
551  a.resize(n);
552  for (int i=0; i<n; i++)
553  a[i] = 0;
554  int k = 1;
555  int y = n-1;
556  std::vector<std::vector<uint32_t>> tmp_poss_perms;
557  while (k != 0 && (sectors_max > poss_perms[s_i].size() || !sectors_max)) {
558  int x = a[k-1] + 1;
559  k -= 1;
560  while (2*x <= y) {
561  a[k] = x;
562  y -= x;
563  k += 1;
564  }
565  int l = k+1;
566  while (x <= y && (sectors_max > poss_perms[s_i].size() || !sectors_max)) {
567  a[k] = x;
568  a[l] = y;
569  std::vector<uint32_t> b;
570  for (std::vector<int>::size_type j=0; j!=k+2; j++)
571  b.push_back(a[j]);
572  tmp_poss_perms.push_back(b);
573  x += 1;
574  y -= 1;
575  }
