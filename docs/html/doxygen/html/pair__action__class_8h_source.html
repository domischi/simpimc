50  void GenerateParticlePairs(const std::vector<std::pair<uint32_t,uint32_t>> &particles, std::vector<uint32_t> &particles_a, std::vector<uint32_t> &particles_b, std::vector<std::pair<uint32_t,uint32_t>> &particle_pairs)
51  {
52  // Make sure particles are of species A or B and organize them accordingly.
53  // If none are A or B, return.
54  for (auto& p: particles) {
55  uint32_t s_i = p.first;
56  uint32_t p_i = p.second;
57  if (s_i == species_a_i)
58  particles_a.push_back(p_i);
59  else if (s_i == species_b_i)
60  particles_b.push_back(p_i);
61  }
62  uint32_t n_a(particles_a.size()), n_b(particles_b.size());
63  if (n_a==0 && n_b==0)
64  return;
65 
66  // Make std::vectors of other particles of species A
67  std::vector<uint32_t> other_particles_a;
68  for (uint32_t p_i=0; p_i<path.species_list[species_a_i]->n_part; ++p_i) {
69  if (find(particles_a.begin(), particles_a.end(), p_i)==particles_a.end())
70  other_particles_a.push_back(p_i);
71  }
72  // Make std::vectors of other particles of species B
73  std::vector<uint32_t> other_particles_b;
74  for (uint32_t p_i=0; p_i<path.species_list[species_b_i]->n_part; ++p_i) {
75  if (find(particles_b.begin(), particles_b.end(), p_i)==particles_b.end())
76  other_particles_b.push_back(p_i);
77  }
78 
79  // Homologous
80  if (species_a_i == species_b_i) {
81  // Loop over A particles with other A particles
82  for (auto& p: particles_a)
83  for (auto& q: other_particles_a)
84  particle_pairs.push_back(std::make_pair(p,q));
85  // Loop over A particles with A particles
86  for (uint32_t p=0; p<n_a-1; ++p)
87  for (uint32_t q=p+1; q<n_a; ++q)
88  particle_pairs.push_back(std::make_pair(particles_a[p],particles_a[q]));
89  // Heterologous
90  } else {
91  // Loop over A particles with other B particles
92  for (auto& p: particles_a)
93  for (auto& q: other_particles_b)
94  particle_pairs.push_back(std::make_pair(p,q));
95  // Loop B particles with other A particles

126  virtual vec<double> CalcGradientU(const uint32_t b_i, const uint32_t b_j, const uint32_t p_i, const uint32_t p_j, const uint32_t level)
127  {
128  // Store original position for particle i
129  std::shared_ptr<Bead> b(path(species_a_i,p_i,b_i));
130  vec<double> r0(b->r);
131 
132  // Numerical tolerance
133  double eps = 1.e-7;
134 
135  // Calculate numerical gradient
136  double r_mag, r_p_mag, r_r_p_mag;
137  vec<double> tot(path.n_d);
138  for (uint32_t d_i=0; d_i<path.n_d; ++d_i) {
139  b->r(d_i) = r0(d_i) + eps;
140  path.DrDrpDrrp(b_i,b_j,species_a_i,species_b_i,p_i,p_j,r_mag,r_p_mag,r_r_p_mag);
141  double f1 = CalcU(r_mag,r_p_mag,r_r_p_mag,level);
142  b->r(d_i) = r0(d_i) - eps;
143  path.DrDrpDrrp(b_i,b_j,species_a_i,species_b_i,p_i,p_j,r_mag,r_p_mag,r_r_p_mag);
144  double f2 = CalcU(r_mag,r_p_mag,r_r_p_mag,level);
145  tot(d_i) = (f1-f2)/(2.*eps);
146  b->r(d_i) = r0(d_i);
147  }
148 
149  return tot;
150  }

284  virtual double GetAction(const uint32_t b_0, const uint32_t b_1, const std::vector<std::pair<uint32_t,uint32_t>> &particles, const uint32_t level)
285  {
286  // Return zero if not relevant
287  if (level > max_level || is_constant || species_a_i < 0 || species_b_i < 0)
288  return 0.;
289 
290  // Generate particle pairs
291  std::vector<uint32_t> particles_a, particles_b;
292  std::vector<std::pair<uint32_t,uint32_t>> particle_pairs;
293  GenerateParticlePairs(particles, particles_a, particles_b, particle_pairs);
294  if (particle_pairs.size() == 0)
295  return 0.;
296 
297  // Sum up contributing terms
298  uint32_t skip = 1<<level;
299  double tot = 0.;
300  for (uint32_t b_i=b_0; b_i<b_1; b_i+=skip) {
301  uint32_t b_j = b_i+skip;
302  uint32_t b_k = b_i-skip+path.n_bead;
303  for (auto& particle_pair: particle_pairs) {
304  double r_mag, r_p_mag, r_r_p_mag;
305  path.DrDrpDrrp(b_i,b_j,species_a_i,species_b_i,particle_pair.first,particle_pair.second,r_mag,r_p_mag,r_r_p_mag);
306  tot += CalcU(r_mag,r_p_mag,r_r_p_mag,level);
307  }
308  }
309 
310  // Add in long range part
311  if (use_long_range) { // FIXME: currently this assumes level = 0
312  if (path.species_list[species_a_i]->need_update_rho_k && path.GetMode()==NEW_MODE) {
313  path.UpdateRhoKP(b_0, b_1, species_a_i, particles_a, level);
314  path.species_list[species_a_i]->need_update_rho_k = false;
315  }
316  if (path.species_list[species_b_i]->need_update_rho_k && path.GetMode()==NEW_MODE) {
317  path.UpdateRhoKP(b_0, b_1, species_b_i, particles_b, level);
318  path.species_list[species_b_i]->need_update_rho_k = false;
319  }
320  tot += CalcULong(b_0, b_1, level);
321  }
322 
323  return tot;
324  }
