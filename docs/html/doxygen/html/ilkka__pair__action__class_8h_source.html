182  virtual vec<double> CalcGradientU(const uint32_t b_i, const uint32_t b_j, const uint32_t p_i, const uint32_t p_j, const uint32_t level)
183  {
184  // Constants
185  vec<double> r, r_p, r_r_p;
186  double r_mag, r_p_mag, r_r_p_mag;
187  path.DrDrpDrrp(b_i,b_j,species_a_i,species_b_i,p_i,p_j,r_mag,r_p_mag,r_r_p_mag,r,r_p,r_r_p);
188  double q = 0.5*(r_mag + r_p_mag);
189  double x = q + 0.5*r_r_p_mag;
190  double y = q - 0.5*r_r_p_mag;
191 
192  // Calculate U and (dU/dxp,dU/dyp)
193  double u = 0.;
194  vec<double> du_dxp_du_dyp(2);
195  eval_NUBspline_2d_d_vg(u_xy_spline,x,y,&u,du_dxp_du_dyp.memptr());
196 
197  // Do chain rule to get (dU/dx_i,dU/dy_i,dU/dz_i)
198  // dU/dx_i = dU/dxp dxp/dx_i + dU/dyp dyp/dx_i
199  // and same for y_i and z_i
200  //
201  // dxp/dx_i = d(q+0.5*s)/dx_i where q = 0.5*(|r| + |r_p|) and s = |r-r_p|
202  // with r = r_i-r_j and r_p = r_i'-r_j'
203  // d|r|/dx_i = d(sqrt((x_i-x_j)^2 + (y_i-y_j)^2 + (z_i-z_j)^2))/dx_i
204  // = (x_i-x_j)/|r|
205  // d|r_p|/dx_i = 0
206  // d|r-r_p|/dx_i = d(sqrt((x_i-x_j-x_i'+x_j')^2 + (y_i-y_j-y_i'+y_j')^2 + (z_i-z_j-z_i'+z_j')^2))/dx_i
207  // = (x_i-x_j-x_i'+x_j')/|r-r_p|
208  // => dxp/dx = 0.5*((x_i-x_j)/|r| + (x_i-x_j-x_i'+x_j')/|r-r_p|)
209  // dyp/dx = 0.5*((x_i-x_j)/|r| - (x_i-x_j-x_i'+x_j')/|r-r_p|)
210  // => dU/dx = 0.5*[dU/dxp*((x_i-x_j)/|r| + (x_i-x_j-x_i'+x_j')/|r-r_p|) + dU/dyp*((x_i-x_j)/|r| - (x_i-x_j-x_i'+x_j')/|r-r_p|)]
211  // => (dU/dx,dU/dy,dU/dz) = 0.5*[dU/dxp*(r/|r| + (r-r_p)/|r-r_p|) + dU/dyp*(r/|r| - (r-r_p)/|r-r_p|)]
212 
213  // Calculate numerical gradient
214  vec<double> r_i_r_mag(r/r_mag), r_r_p_i_r_r_p_mag(r_r_p/r_r_p_mag);
215  if (r_mag == 0.)
216  r_i_r_mag.zeros();
217  if (r_r_p_mag == 0.)
218  r_r_p_i_r_r_p_mag.zeros();
219  vec<double> tot = -0.5*(du_dxp_du_dyp(0)*(r_i_r_mag + r_r_p_i_r_r_p_mag) + du_dxp_du_dyp(1)*(r_i_r_mag - r_r_p_i_r_r_p_mag));
220 
221  // Subtract out long range part
222  if (use_long_range) {
223  // Limits
224  SetLimits(r_u_min, r_u_max, r_mag, r_p_mag);
225 
226  // Splines
227  double tmp_u, tmp_du_dr;
228  eval_NUBspline_1d_d_vg(u_long_r_spline,r_mag,&tmp_u,&tmp_du_dr);
229  tot -= 0.5*tmp_du_dr*r_i_r_mag;
230  }
231 
232  return tot;
233  }
