36  {
37  // Update t
38  UpdatePermTable();
39  mat<double> t_c = t;
40 
41  // Choose particles
42  uint32_t p0 = rng.UnifRand(n_part) - 1; // Pick first particle at random
43  uint32_t p = p0;
44  std::vector<uint32_t> ps;
45  uint32_t n_perm = 0;
46  do {
47  // Add particle to ps
48  ps.push_back(p);
49  n_perm = ps.size();
50 
51  // Make sure returning to previous particles is not an option
52  for (uint32_t i=0; i<n_perm; ++i)
53  t_c(p,ps[i]) = 0.;
54 
55  // Allow cycle to close
56  if (n_perm > 0)
57  t_c(p,p0) = t(p,p0);
58 
59  // Disallow odd permutations (even number of particles) for fixed-node fermions
60  bool is_fixed_node_odd_perm = path.species_list[species_i]->fermi && path.species_list[species_i]->fixed_node && !(n_perm%2);
61  if (is_fixed_node_odd_perm)
62  t_c(p,p0) = 0.;
63 
64  // Calculate row total
65  double Q_p = 0.;
66  double Q_p_c = 0.;
67  for (uint32_t i=0; i<n_part; ++i) {
68  Q_p += t(p,i);
69  Q_p_c += t_c(p,i);
70  }
71 
72  // Decide whether or not to continue
73  if (Q_p_c/Q_p < rng.UnifRand()) {
74  c.type = n_perm - 1;
75  return 0;
76  }
77 
78  // Select next particle with bisective search
79  double x = rng.UnifRand();
80  double t_Q = 0.;
81  for (uint32_t i=0; i<n_part; ++i) { // TODO: not doing bisection
82  t_Q += t_c(p,i)/Q_p_c;
83  if (t_Q > x) {
84  p = i;
85  break;
86  }
87  }
88 
89  } while (p != p0);
90 
91  // Set weight
92  c.weight = 1.;
93  for (uint32_t i=0; i<n_perm-1; ++i)
94  c.weight *= t(ps[i],ps[i+1])/t(ps[i],ps[i]);
95  c.weight *= t(ps[n_perm-1],ps[0])/t(ps[n_perm-1],ps[n_perm-1]);
96 
97  // Set particles
98  c.part.set_size(n_perm);
99  for (uint32_t i=0; i<n_perm; ++i)
100  c.part(i) = ps[i];
101  c.type = n_perm-1;
102 
103  // Set perms
104  c.perm.set_size(n_perm);
105  for (uint32_t i=0; i<n_perm-1; ++i)
106  c.perm(i) = i+1;
107  c.perm(n_perm-1) = 0;
108  c.i_perm.set_size(n_perm);
109  c.i_perm(0) = n_perm-1;
110  for (uint32_t i=1; i<n_perm; ++i)
111  c.i_perm(i) = i-1;
112

124  (roll_over && path.bead_loop[bead1]>=path.ref_bead));
125  if (include_ref)

131  if (!SelectCycleIterative(c)) {

133  return 0; // do not attempt bisection since permutation not accepted
134  }

137  // Set up pointers
138  std::vector<std::pair<uint32_t,uint32_t>> particles;
139  field< std::shared_ptr<Bead>> bead_i(n_perm_part), bead_fm1(n_perm_part), bead_f(n_perm_part);
140  for (uint32_t i=0; i<n_perm_part; i++) {

142  bead_fm1(i) = bead_i(i)->NextB(n_bisect_beads-1);
143  bead_f(i) = bead_fm1(i)->next;
144  particles.push_back(std::make_pair(species_i,c.part(i)));
145  }
146  if (roll_over) {

148  particles.push_back(std::make_pair(species_i,bead_f(i)->p));
149  sort(particles.begin(), particles.end());

162  }

166  double prev_action_change = -log(c.weight);
167  for (int level_i = n_level-1; level_i >= 0; level_i -= 1) {
168 
169  // Level specific quantities
170  uint32_t skip = 1<<level_i;
171  double sigma = sqrt(path.tau*skip*lambda);
172 
173  // Calculate sampling probability
174  double old_log_sample_prob = 0.;
175  double new_log_sample_prob = 0.;
176  for (uint32_t i=0; i<n_perm_part; i++) {
177  bead_a(i) = bead_i(i);
178  while(bead_a(i) != bead_f(i)) {
179  // Old sampling
180  path.SetMode(OLD_MODE);
181  bead_b(i) = path.GetNextBead(bead_a(i),skip);
182  bead_c(i) = path.GetNextBead(bead_b(i),skip);
183  vec<double> delta_old(path.Dr(bead_b(i),path.RBar(bead_c(i),bead_a(i))));
184  old_log_sample_prob += rho_free_splines[level_i].GetLogRhoFree(delta_old);
185 
186  // New sampling
187  path.SetMode(NEW_MODE);
188  bead_b(i) = path.GetNextBead(bead_a(i),skip);
189  bead_c(i) = path.GetNextBead(bead_b(i),skip);
190  vec<double> delta_new(path.n_d);
191  rng.NormRand(delta_new, 0., sigma);
192  path.PutInBox(delta_new);
193  bead_b(i)->r = path.RBar(bead_c(i),bead_a(i)) + delta_new;
194  new_log_sample_prob += rho_free_splines[level_i].GetLogRhoFree(delta_new);
195 
196  bead_a(i) = bead_c(i);
197  }
198  }
199 
200  // Calculate action change
201  double old_action = 0.;
202  double new_action = 0.;
203  for (auto& action: action_list) {
204  // Old action
205  path.SetMode(OLD_MODE);
206  old_action += action->GetAction(bead0, bead1, particles, level_i);
207 
208  // New action
209  path.SetMode(NEW_MODE);
210  new_action += action->GetAction(bead0, bead1, particles, level_i);
211  }
212 
213  // Calculate acceptance ratio
214  double log_sample_ratio = -new_log_sample_prob + old_log_sample_prob;
215  double current_action_change = new_action - old_action;
216  double log_accept_probability = log_sample_ratio - current_action_change + prev_action_change;
217 
218  // Metropolis step
219  if (log_accept_probability < log(rng.UnifRand()))
220  return 0;
221 
222  prev_action_change = current_action_change;
223  }
224 
225  if (include_ref)
226  ref_accept++;
227 
228  return 1;
