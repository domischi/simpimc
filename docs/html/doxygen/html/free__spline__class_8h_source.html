27  FreeSpline(const double L, const uint32_t n_images, const double lambda, const double tau, const bool use_tau_derivative)
28  {
29  // Make constants
30  i_4_lambda_tau = 1./(4.*lambda*tau);
31  i_4_lambda_tau_tau = 1./(4.*lambda*tau*tau);
32 
33  // Setup grid
34  Ugrid r_grid;
35  double t_l = L;
36  if (L == 0.)
37  t_l = 1000.;
38  r_grid.start = -t_l/2.;
39  r_grid.end = t_l/2.;
40  r_grid.num = 10000;
41  double dr = (r_grid.end - r_grid.start)/(r_grid.num - 1);
42 
43  // Make image_action, d_image_action_d_tau, d_image_action_d_r
44  vec<double> image_action(zeros<vec<double>>(r_grid.num));
45  vec<double> d_image_action_d_tau(zeros<vec<double>>(r_grid.num));
46  for (uint32_t i=0; i<r_grid.num; ++i) {
47  double r = r_grid.start + i*dr;
48  double r_2 = r*r;
49  double r_2_i_4_lambda_tau = r_2*i_4_lambda_tau;
50  for (uint32_t image=1; image<=n_images; image++) {
51  double r_p(r+image*t_l);
52  double r_p_2_i_4_lambda_tau = r_p*r_p*i_4_lambda_tau;
53  double d_r_2_r_p_2_i_4_lambda_tau = r_2_i_4_lambda_tau - r_p_2_i_4_lambda_tau;
54  double exp_part_p = exp(d_r_2_r_p_2_i_4_lambda_tau);
55  double r_m(r-image*t_l);
56  double r_m_2_i_4_lambda_tau = r_m*r_m*i_4_lambda_tau;
57  double d_r_2_r_m_2_i_4_lambda_tau = r_2_i_4_lambda_tau - r_m_2_i_4_lambda_tau;
58  double exp_part_m = exp(d_r_2_r_m_2_i_4_lambda_tau);
59  image_action(i) += exp_part_p + exp_part_m;
60  if (use_tau_derivative)
61  d_image_action_d_tau(i) += (d_r_2_r_p_2_i_4_lambda_tau*exp_part_p + d_r_2_r_m_2_i_4_lambda_tau*exp_part_m)/tau;
62  }
63  if (use_tau_derivative)
64  d_image_action_d_tau(i) = d_image_action_d_tau(i)/(1.+image_action(i));
65  image_action(i) = -log1p(image_action(i));
66  }
67  BCtype_d xBC = {NATURAL, NATURAL};
68  image_action_spline = create_UBspline_1d_d(r_grid, xBC, image_action.memptr());
69  if (use_tau_derivative)
70  d_image_action_d_tau_spline = create_UBspline_1d_d(r_grid, xBC, d_image_action_d_tau.memptr());
71  }
